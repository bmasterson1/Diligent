      ```python
      import re

      def _slugify(text: str, max_length: int = 40) -> str:
          slug = text.lower()
          slug = re.sub(r"[^a-z0-9]+", "-", slug)
          slug = slug.strip("-")
          if len(slug) > max_length:
              slug = slug[:max_length].rstrip("-")
          return slug
      ```

   c. Next task ID helper:
      ```python
      def _next_task_id(tasks_dir: Path) -> int:
          max_num = 0
          if tasks_dir.exists():
              pattern = re.compile(r"^(\d{3})-")
              for child in tasks_dir.iterdir():
                  if child.is_dir():
                      m = pattern.match(child.name)
                      if m:
                          num = int(m.group(1))
                          if num > max_num:
                              max_num = num
          return max_num + 1
      ```

   d. Task directory reader helper (for list/complete):
      ```python
      def _read_tasks(tasks_dir: Path) -> list[dict]:
          tasks = []
          if not tasks_dir.exists():
              return tasks
          pattern = re.compile(r"^(\d{3})-")
          for child in sorted(tasks_dir.iterdir()):
              if child.is_dir() and pattern.match(child.name):
                  status_file = child / "status.yaml"
                  if status_file.exists():
                      data = yaml.safe_load(status_file.read_text(encoding="utf-8"))
                      tasks.append({
                          "id": child.name[:3],
                          "dir_name": child.name,
                          "description": data.get("description", ""),
                          "status": data.get("status", "open"),
                          "created": data.get("created", ""),
                          "path": str(child),
                      })
          return tasks
      ```
