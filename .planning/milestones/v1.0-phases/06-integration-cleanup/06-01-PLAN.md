```python
def _find_diligence_dir(env_cwd: Optional[str] = None) -> Path:
    """Locate the .diligence/ directory.
    Checks DILIGENT_CWD env override first (for testing), then walks up from cwd.
    Raises: click.ClickException: If .diligence/ not found.
    """
    if env_cwd:
        candidate = Path(env_cwd) / ".diligence"
        if candidate.is_dir():
            return candidate
    cwd = Path.cwd()
    for parent in [cwd] + list(cwd.parents):
        candidate = parent / ".diligence"
        if candidate.is_dir():
            return candidate
    raise click.ClickException(
        "No .diligence/ directory found. Run 'diligent init' first."
    )
```
