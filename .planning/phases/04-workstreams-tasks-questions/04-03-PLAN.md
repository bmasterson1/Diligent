```python
def _next_question_id(questions: QuestionsFile) -> str:
    # Scans Q-NNN IDs, returns max+1 formatted as Q-{NNN:03d}
    max_num = 0
    for q in questions.questions:
        if q.id.startswith("Q-"):
            try:
                num = int(q.id[2:])
                if num > max_num:
                    max_num = num
            except ValueError:
                pass
    return f"Q-{max_num + 1:03d}"
```
